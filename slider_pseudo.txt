class Blob_Window:
  def __init__(self, window_name="TrackedBars", size=(640, 240)):
    self.window = cv2.namedWindow(window_name)
    cv2.resizeWindow(window_name, *size)
    
    # Define default slider values
    self.hue_min = 0
    self.hue_max = 179
    self.sat_min = 0
    self.sat_max = 255
    self.val_min = 0
    self.val_max = 255
    
    # Construct Trackbar Objects
    cv2.createTrackbar("Hue Min", "TrackedBars", 0, 179, self.on_trackbar)
    cv2.createTrackbar("Hue Max", "TrackedBars", 179, 179, self.on_trackbar)
    cv2.createTrackbar("Sat Min", "TrackedBars", 0, 255, self.on_trackbar)
    cv2.createTrackbar("Sat Max", "TrackedBars", 255, 255, self.on_trackbar)
    cv2.createTrackbar("Val Min", "TrackedBars", 0, 255, self.on_trackbar)
    cv2.createTrackbar("Val Max", "TrackedBars", 255, 255, self.on_trackbar)
		
		# Instantiate Sliders On Runtime
		self.update_masks(0)


  def on_trackbar(val):
    self.hue_min = cv2.getTrackbarPos("Hue Min", "TrackedBars")		
    self.hue_max = cv2.getTrackbarPos("Hue Max", "TrackedBars")
    self.sat_min = cv2.getTrackbarPos("Sat Min", "TrackedBars")
    self.sat_max = cv2.getTrackbarPos("Sat Max", "TrackedBars")
    self.val_min = cv2.getTrackbarPos("Val Min", "TrackedBars")
    self.val_max = cv2.getTrackbarPos("Val Max", "TrackedBars")
